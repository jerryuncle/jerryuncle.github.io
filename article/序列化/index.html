==MEMORY_ONLY和MEMORY_ONLY_SER最为常用==

https://blog.csdn.net/houmou/article/details/52491419
cache调用的persist
cache只有一个默认的缓存级别MEMORY_ONLY ，而persist可以根据情况设置其它的缓存级

scala> val rdd = sc.textFile(“file:///home/wzj/data/test.txt”)
rdd: org.apache.spark.rdd.RDD[String] = file:///home/wzj/data/test.txt MapPartitionsRDD[1] at textFile at :24

scala> rdd.collect
res0: Array[String] = Array(study,study,study, no,no, i)

scala> rdd.unpersist()
res5: rdd.type = file:///home/wzj/data/test.txt MapPartitionsRDD[1] at textFile at :24

scala> import org.apache.spark.storage.StorageLevel
import org.apache.spark.storage.StorageLevel

scala> rdd.persist(StorageLevel.MEMORY_ONLY)
res7: rdd.type = file:///home/wzj/data/test.txt MapPartitionsRDD[1] at textFile at :24

scala> rdd.count
res8: Long = 3

scala> rdd.persist(StorageLevel.MEMORY_ONLY_SER)
res14: rdd.type = file:///home/wzj/data/test.txt MapPartitionsRDD[1] at textFile at :24

scala> rdd.count
res15: Long = 3
