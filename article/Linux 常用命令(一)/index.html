1.查看当前光标所在目录

[jumproot@bi_new home]$ pwd
/home

2.切换目录

[jumproot@bi_new /]$ cd home
[jumproot@bi_new home]$ pwd
/home

3.回退家目录方式

3.1 [root@bogon home]# cd
[root@bogon ~]# pwd
/root
3.2 [root@bogon home]# cd ~
[root@bogon ~]# pwd
/root
3.3 [root@bogon home]# cd /root
[root@bogon ~]# pwd
/root

4.回退前一次目录

[root@bogon ~]# cd /home/wzj/
[root@bogon wzj]# cd /home
[root@bogon home]# cd -
/home/wzj

5.返回上一层目录

[root@bogon wzj]# cd …
[root@bogon home]#

7.创建文件夹

[root@bogon home]# cd wzj
[root@bogon wzj]# ll
total 0
[root@bogon wzj]# mkdir a
[root@bogon wzj]# ls
a
[root@bogon wzj]# mkdir -p b/c/d
[root@bogon wzj]# ls
a b

8.文件拷贝和移动

mv移动 始终只有1份 cp拷贝 原文件不动，新增一份
[root@bogon a]# ll
total 4
-rw-r–r--. 1 root root 12 Nov 13 12:50 test.log
[root@bogon a]# cd …/b/
[root@bogon b]# ls
c
[root@bogon b]# cd …/a
[root@bogon a]# mv test.log …/b/
[root@bogon a]# cd …/b
[root@bogon b]# ls
c test.log

9.创建空文件

[root@bogon b]# ls
c test.log
[root@bogon b]# touch test2.log
[root@bogon b]# ls
c test2.log test.log

10.查看日志

grep 过滤
cat 文件内容一下子全部显示
more 文件内容一页一页的往下翻，按空格键往下，回退不了，按q退出
less 文件内容 往下 往上 按上下箭头的按键 按q键退出
实时查看
tail -f xxx.log 当日志有预警大小发生移动和重新生成之后内容无法监测
tail -F xxx.log -F相当于-f+retry

实例：查找日志
cat xxx.log | grep -A 10 ERROR 后10行
cat xxx.log | grep -B 10 ERROR 前10行
cat xxx.log | grep -C 30 ERROR 前后各30行 迅速定位ERROR上下文
tail -300f messages 实时查看倒数300行文件
tail -n +10 test.log 查询文件第10行开始以后的所有日志
head -n -10 test.log 查询文件尾部10行以前的所以日志
